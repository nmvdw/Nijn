Require Import Nijn.Nijn.

Open Scope poly_scope.

The base types
Inductive base_types :=
| TBtype.

Global Instance decEq_base_types : decEq base_types.
Proof.
  decEq_finite.
Defined.

Definition Btype := Base TBtype.

The function symbols and their arities
Inductive fun_symbols :=
| TF
| TG.

Global Instance decEq_fun_symbols : decEq fun_symbols.
Proof.
  decEq_finite.
Defined.

Definition arities f
  := match f with
     | TF => Btype ⟶ Btype
     | TG => Btype
     end.

Definition F {C} : tm arities C _ := BaseTm TF.
Definition G {C} : tm arities C _ := BaseTm TG.

The rewrite rules
Program Definition rew_1 := make_rewrite ∙ _ (F · G) (F · (F · G)).

The AFS
Definition not_SN_afs :=
  make_afs
    arities
    (rew_1 :: nil).

Strong normalization
Definition not_SN_not_SN : ~(isSN not_SN_afs).
Proof.
  apply inc_loop_to_notSN.
  simple refine {| inc_loop_con := ∙ ; inc_loop_ty := Btype |}.
  - exact (F · G).
  - exact (F · (F · G)).
  - refine (rew_rewrite_rule not_SN_afs (MakeMem rew_1 _) (idSub _ _)).
    solve_In.
  - pose (dec_subTerm (F · G : tm _ ∙ _) (F · (F · G) : tm _ ∙ _)).
    cbn in s.
    apply SubAppR.
    simple refine (EqSubTm _ _ _ _ _).
    + reflexivity.
    + reflexivity.
    + reflexivity.
Qed.
